// Required HTTP headers (set on your web server)

# ASP.NET Core Minimal API — add to Program.cs

app.Use(async (ctx, next) =>
{
    var url = "http://localhost:5000/csp-report";
    ctx.Response.Headers.Append("Reporting-Endpoints",
        $"csp-endpoint=\"{url}\"");
    ctx.Response.Headers.Append("Content-Security-Policy",
        "default-src 'self'; " +
        "script-src 'self'; " +
        "style-src 'self' https://fonts.googleapis.com; " +
        "font-src 'self' https://fonts.gstatic.com; " +
        "img-src 'self'; " +
        "report-to csp-endpoint; " +
        "report-uri /csp-report");   // fallback for older browsers
    await next();
});
